## 3. Установка пакетов

### Debian/Ubuntu (apt)

Bash

```
# ═══════════════════════════════════════════════════════════
#                    APT (DEBIAN/UBUNTU)
# ═══════════════════════════════════════════════════════════

# Обновить СПИСОК доступных пакетов (не устанавливает!)
sudo apt update

# Обновить ВСЕ установленные пакеты
sudo apt upgrade

# Установить пакет
sudo apt install nginx

# Установить несколько пакетов
sudo apt install nginx vim curl

# Установить без подтверждения (для скриптов)
sudo apt install -y nginx

# Установить конкретную версию
sudo apt install nginx=1.18.0-0ubuntu1

# ───────────────────────────────────────────────────────────
# Удаление
# ───────────────────────────────────────────────────────────

# Удалить пакет (оставить конфигурацию)
sudo apt remove nginx

# Удалить пакет + конфигурацию
sudo apt purge nginx

# Удалить неиспользуемые зависимости
sudo apt autoremove

# ───────────────────────────────────────────────────────────
# Поиск и информация
# ───────────────────────────────────────────────────────────

# Поиск пакета
apt search nginx

# Информация о пакете
apt show nginx

# Список установленных пакетов
apt list --installed

# Какой пакет содержит файл
dpkg -S /usr/sbin/nginx

# Какие файлы в пакете
dpkg -L nginx

# ───────────────────────────────────────────────────────────
# Установка из файла
# ───────────────────────────────────────────────────────────

# Установить .deb файл
sudo dpkg -i package.deb

# Если ошибка зависимостей — исправить
sudo apt install -f
```

### RHEL/CentOS/Fedora (dnf)

Bash

```
# ═══════════════════════════════════════════════════════════
#                    DNF (RHEL/CENTOS/FEDORA)
# ═══════════════════════════════════════════════════════════

# DNF — современный пакетный менеджер (RHEL 8+, Fedora)
# YUM — устаревший (RHEL 7, CentOS 7)

# Обновить ВСЕ пакеты
sudo dnf upgrade

# Установить пакет
sudo dnf install nginx

# Установить без подтверждения
sudo dnf install -y nginx

# ───────────────────────────────────────────────────────────
# Удаление
# ───────────────────────────────────────────────────────────

# Удалить пакет
sudo dnf remove nginx

# Удалить с неиспользуемыми зависимостями
sudo dnf autoremove

# ───────────────────────────────────────────────────────────
# Поиск и информация
# ───────────────────────────────────────────────────────────

# Поиск
dnf search nginx

# Информация
dnf info nginx

# Какой пакет предоставляет файл
dnf provides /usr/sbin/nginx

# ───────────────────────────────────────────────────────────
# Установка из файла
# ───────────────────────────────────────────────────────────

# Установить .rpm файл
sudo dnf install ./package.rpm
# или
sudo rpm -ivh package.rpm
```

## 4. Работа с файлами и директориями

### Просмотр содержимого

Bash

```
# ═══════════════════════════════════════════════════════════
#                ПРОСМОТР ФАЙЛОВ И ДИРЕКТОРИЙ
# ═══════════════════════════════════════════════════════════

# Простой список
ls

# Подробный формат
ls -l

# Включая СКРЫТЫЕ файлы (начинаются с точки)

ls -lh

# Сортировка по размеру
ls -lhS

# Сортировка по дате
ls -lt

# Рекурсивно (с поддиректориями)
ls -R
```

### Размер файлов

Bash

```
# ═══════════════════════════════════════════════════════════
#                      РАЗМЕР ФАЙЛОВ
# ═══════════════════════════════════════════════════════════

# Размер конкретной директории (суммарно)
du -sh /var/log
# -s = summary (итого)
# -h = human-readable

# Размер каждого элемента в директории
du -sh *

# Размер в мегабайтах
du -sm *

# С сортировкой по убыванию
du -sh * | sort -rh

# Топ-10 самых больших
du -ah | sort -rh | head -10

# Только первый уровень вложенности
du -h --max-depth=1 /var
```

### Навигация и базовые операции

Bash

```
# ═══════════════════════════════════════════════════════════
#                НАВИГАЦИЯ И ОПЕРАЦИИ
# ═══════════════════════════════════════════════════════════

# Текущая директория
pwd

# Перейти в директорию
cd /var/log

# Домашняя директория
cd ~
cd

# Предыдущая директория
cd -

# На уровень выше
cd ..

# ───────────────────────────────────────────────────────────
# Создание
# ───────────────────────────────────────────────────────────

# Создать директорию
mkdir mydir

# Создать вложенные директории
mkdir -p parent/child/grandchild

# Создать пустой файл
touch file.txt

# ───────────────────────────────────────────────────────────
# Копирование и перемещение
# ───────────────────────────────────────────────────────────

# Копировать файл
cp source.txt dest.txt

# Копировать директорию рекурсивно
cp -r source_dir/ dest_dir/

# Переместить/переименовать
mv old.txt new.txt
mv file.txt /path/to/destination/

# ───────────────────────────────────────────────────────────
# Удаление
# ───────────────────────────────────────────────────────────

# Удалить файл
rm file.txt

# Удалить директорию рекурсивно
rm -r directory/

# Без подтверждения (ОСТОРОЖНО!)
rm -rf directory/

# ───────────────────────────────────────────────────────────
# Просмотр содержимого файлов
# ───────────────────────────────────────────────────────────

# Вывести весь файл
cat file.txt

# Постраничный просмотр
less file.txt
# (q — выход, / — поиск)

# Первые/последние строки
head file.txt
head -n 20 file.txt
tail file.txt
tail -n 20 file.txt

# Следить за файлом в реальном времени
tail -f /var/log/syslog
```

## 5. Pipes и перенаправление

### Pipe (конвейер)

Bash

```
# ═══════════════════════════════════════════════════════════
#                    PIPE (КОНВЕЙЕР)
# ═══════════════════════════════════════════════════════════

# | — передаёт STDOUT одной команды на STDIN другой

# Найти строку в выводе
ps aux | grep nginx

# Посчитать количество строк
ls -1 | wc -l

# Отсортировать и убрать дубликаты
cat file.txt | sort | uniq

# Цепочка из нескольких команд
cat access.log | grep "404" | awk '{print $7}' | sort | uniq -c | sort -rn

# Постраничный просмотр
ls -la /etc | less
```

### Перенаправление в файл

Bash

```
# ═══════════════════════════════════════════════════════════
#                ПЕРЕНАПРАВЛЕНИЕ В ФАЙЛ
# ═══════════════════════════════════════════════════════════

# Дескрипторы файлов:
# 0 = stdin  (ввод)
# 1 = stdout (стандартный вывод)
# 2 = stderr (ошибки)

# ───────────────────────────────────────────────────────────
# Перенаправление stdout
# ───────────────────────────────────────────────────────────

# Записать stdout в файл (ПЕРЕЗАПИСАТЬ)
command > output.txt

# Записать stdout в файл (ДОПИСАТЬ)
command >> output.txt

# ───────────────────────────────────────────────────────────
# Перенаправление stderr (ТОЛЬКО ОШИБКИ)
# ───────────────────────────────────────────────────────────

# Записать stderr в файл
command 2> errors.txt

# Дописать stderr в файл
command 2>> errors.txt

# ───────────────────────────────────────────────────────────
# Комбинации
# ───────────────────────────────────────────────────────────

# stdout и stderr в РАЗНЫЕ файлы
command > output.txt 2> errors.txt

# stdout и stderr в ОДИН файл
command > all.txt 2>&1
command &> all.txt        # bash-сокращение

# Отбросить весь вывод
command > /dev/null 2>&1
command &> /dev/null

# ───────────────────────────────────────────────────────────
# Запись от root (sudo не работает с > напрямую)
# ───────────────────────────────────────────────────────────

# НЕПРАВИЛЬНО:
sudo echo "text" > /etc/file    # Ошибка!

# ПРАВИЛЬНО:
echo "text" | sudo tee /etc/file         # перезаписать
echo "text" | sudo tee -a /etc/file      # дописать
```

## 6. Exit Code

text

```
┌────────────────────────────────────────────────────────────┐
│                       EXIT CODE                            │
├────────────────────────────────────────────────────────────┤
│                                                            │
│  Exit code (код завершения) — число 0-255, которое        │
│  каждый процесс возвращает при завершении                 │
│                                                            │
│  ┌───────┬────────────────────────────────────────────┐   │
│  │  Код  │ Значение                                   │   │
│  ├───────┼────────────────────────────────────────────┤   │
│  │   0   │ УСПЕХ                                      │   │
│  │   1   │ Общая ошибка                               │   │
│  │   2   │ Неправильное использование команды         │   │
│  │  126  │ Файл не является исполняемым               │   │
│  │  127  │ Команда не найдена                         │   │
│  │  128  │ Неверный аргумент exit                     │   │
│  │ 128+N │ Убит сигналом N                            │   │
│  │  130  │ Прерван Ctrl+C (128+2=SIGINT)              │   │
│  │  137  │ Убит kill -9 (128+9=SIGKILL)               │   │
│  │  143  │ Завершён SIGTERM (128+15)                  │   │
│  └───────┴────────────────────────────────────────────┘   │
│                                                            │
└────────────────────────────────────────────────────────────┘
```

Bash

```
# Узнать exit code последней команды
echo $?

# Примеры
ls /tmp
echo $?           # 0 (успех)

ls /несуществующая
echo $?           # 2 (ошибка)

# ───────────────────────────────────────────────────────────
# Условное выполнение
# ───────────────────────────────────────────────────────────

# && — выполнить если ПРЕДЫДУЩАЯ УСПЕШНА (exit code = 0)
apt update && apt install nginx
mkdir /tmp/test && cd /tmp/test

# || — выполнить если ПРЕДЫДУЩАЯ НЕУСПЕШНА (exit code ≠ 0)
cd /nonexistent || echo "Директория не найдена"
ping -c1 server || echo "Сервер недоступен"

# Комбинация
command && echo "OK" || echo "FAIL"

# ; — выполнить ВСЕГДА, независимо от результата
echo "start" ; command ; echo "end"
```

## 7. Linux File Hierarchy Structure (FHS)

text

```
/                        Корень файловой системы
│
├── /bin                 Основные команды (ls, cp, bash)
│                        → обычно symlink на /usr/bin
│
├── /sbin                Системные команды (fdisk, iptables)
│                        → обычно symlink на /usr/sbin
│
├── /etc                 Конфигурации системы
│   ├── passwd           Информация о пользователях
│   ├── shadow           Хеши паролей
│   ├── group            Группы
│   ├── sudoers          Настройки sudo
│   ├── fstab            Точки монтирования
│   ├── hosts            Локальный DNS
│   ├── ssh/             Конфигурация SSH
│   └── apt/             Репозитории (Debian)
│
├── /home                Домашние директории пользователей
│   ├── user1/
│   └── user2/
│
├── /root                Домашняя директория root
│
```

```
#                  СОЗДАНИЕ ПОЛЬЗОВАТЕЛЯ

# С дополнительными группами

# Задать пароль

# Неинтерактивно задать пароль (для скриптов)
```

### Управление группами

Bash

```
# ═══════════════════════════════════════════════════════════
#                  УПРАВЛЕНИЕ ГРУППАМИ
# ═══════════════════════════════════════════════════════════

# Посмотреть группы пользователя
groups john
id john
# uid=1001(john) gid=1001(john) groups=1001(john),27(sudo),999(docker)

# ───────────────────────────────────────────────────────────
# ДОБАВИТЬ в группу
# ───────────────────────────────────────────────────────────

sudo usermod -aG docker john
# -a = APPEND (добавить, НЕ заменять!)
# -G = дополнительные группы

# ⚠️ ВАЖНО: без -a все группы будут ЗАМЕНЕНЫ!
# ❌ sudo usermod -G docker john    ← ОПАСНО!
# ✅ sudo usermod -aG docker john   ← ПРАВИЛЬНО

# Добавить в несколько групп
sudo usermod -aG docker,sudo,developers john

# ───────────────────────────────────────────────────────────
# ИЗМЕНИТЬ основную группу
# ───────────────────────────────────────────────────────────

sudo usermod -g newgroup john

# ───────────────────────────────────────────────────────────
# УДАЛИТЬ из группы
# ───────────────────────────────────────────────────────────

sudo gpasswd -d john docker

# ───────────────────────────────────────────────────────────
# Управление самими группами
# ───────────────────────────────────────────────────────────

# Создать группу
sudo groupadd developers

# Удалить группу
sudo groupdel developers

# Переименовать группу
sudo groupmod -n newname oldname
```

## 9. Система прав доступа

### Как читать права

```
ls -l file.txt

│ │  │  │    └─────── владелец
│ │  │  └──────────── права для ОСТАЛЬНЫХ (others): r--
│ │  └─────────────── права для ГРУППЫ (group): r-x
│ └────────────────── права для ВЛАДЕЛЬЦА (user): rwx
└──────────────────── тип: - файл, d директория, l ссылка
```

### Что означают права

text

```
┌───────┬──────────────────────┬──────────────────────────┬─────┐
│ Право │ На ФАЙЛЕ             │ На ДИРЕКТОРИИ            │ Бит │
├───────┼──────────────────────┼──────────────────────────┼─────┤
│   r   │ Читать содержимое    │ Видеть список файлов (ls)│  4  │
│   w   │ Изменять содержимое  │ Создавать/удалять файлы  │  2  │
│   x   │ Выполнять как прогр. │ Войти в директорию (cd)  │  1  │
└───────┴──────────────────────┴──────────────────────────┴─────┘
```

### Владелец и группа-владелец

Bash

```
# Владелец — пользователь, которому принадлежит файл
#            (обычно тот, кто создал файл)
#            Может изменять права (chmod)

# Группа-владелец — группа, привязанная к файлу
#                   Все члены группы получают "групповые" права

# Посмотреть владельца и группу
ls -l file.txt
# -rw-r--r-- 1 john developers 1024 Jan 15 file.txt
#               │    │
#               │    └── группа-владелец
#               └─────── владелец
```

### Буквенный формат chmod

Bash

```
# ═══════════════════════════════════════════════════════════
#              БУКВЕННЫЙ ФОРМАТ (СИМВОЛЬНЫЙ)
# ═══════════════════════════════════════════════════════════

# Формат: chmod [кому][оператор][права] файл

# Кому:
# u = user (владелец)
# g = group (группа)
# o = others (остальные)
# a = all (все)

# Оператор:
# + = добавить право
# - = убрать право
# = = установить точно

# Права:
# r, w, x

# ───────────────────────────────────────────────────────────
# Примеры
# ───────────────────────────────────────────────────────────

chmod u+x script.sh           # Добавить execute владельцу
chmod g+rw file.txt           # Добавить read+write группе
chmod o-rwx file.txt          # Убрать всё у остальных
chmod a+r file.txt            # Добавить read всем
chmod u=rwx,g=rx,o=r file.txt # Установить точно
chmod +x script.sh            # Добавить execute всем
```

### Числовой формат chmod

Bash

```
# ═══════════════════════════════════════════════════════════
#              ЧИСЛОВОЙ ФОРМАТ (ВОСЬМЕРИЧНЫЙ)
# ═══════════════════════════════════════════════════════════

# r = 4, w = 2, x = 1
# Право = СУММА битов

# rwx = 4+2+1 = 7
# rw- = 4+2+0 = 6
# r-x = 4+0+1 = 5
# r-- = 4+0+0 = 4
# --- = 0+0+0 = 0

# Три цифры: [user][group][others]

chmod 755 script.sh    # rwxr-xr-x
chmod 644 file.txt     # rw-r--r--
chmod 700 private/     # rwx------
chmod 600 secret.key   # rw-------
chmod 777 file.txt     # rwxrwxrwx (НЕБЕЗОПАСНО!)
```

### Смена владельцев и прав

Bash

```
# ═══════════════════════════════════════════════════════════
#                  СМЕНА ВЛАДЕЛЬЦЕВ
# ═══════════════════════════════════════════════════════════

# Сменить владельца
sudo chown newuser file.txt

# Сменить владельца И группу
sudo chown newuser:newgroup file.txt

# Сменить только группу
sudo chgrp newgroup file.txt
# или
sudo chown :newgroup file.txt

# Рекурсивно (вся директория)
sudo chown -R user:group /path/to/dir/
sudo chmod -R 755 /path/to/dir/
```

## 10. Дистрибутивы для Docker

text

```
┌──────────────┬──────────┬──────────────────────────────────┐
│ Дистрибутив  │  Размер  │ Особенности                      │
├──────────────┼──────────┼──────────────────────────────────┤
│ Alpine       │   ~5 MB  │ Самый популярный для Docker      │
│ Debian Slim  │  ~25 MB  │ glibc, хорошая совместимость     │
│ Ubuntu Min.  │  ~29 MB  │ Привычный apt, много документации│
│ Distroless   │ ~2-20 MB │ Без shell, максимум безопасности │
│ scratch      │    0 MB  │ Пустой образ (только ваш бинарник│
└──────────────┴──────────┴──────────────────────────────────┘

Почему Alpine лидирует:
• Минимальный размер → быстрая загрузка, меньше места
• Мало пакетов → меньше поверхность атаки
• Быстрый пакетный менеджер apk

⚠️ Нюанс: Alpine использует musl libc (вместо glibc)
   → возможны проблемы с Python C-расширениями
   → в таких случаях лучше debian:slim
```

Dockerfile

```
# Alpine
FROM alpine:3.19
RUN apk add --no-cache python3

# Debian Slim (когда нужна glibc-совместимость)
FROM debian:bookworm-slim
RUN apt-get update && apt-get install -y --no-install-recommends python3 \
    && rm -rf /var/lib/apt/lists/*

# Multi-stage + scratch (Go, Rust)
FROM golang:1.22 AS builder
COPY . .
RUN CGO_ENABLED=0 go build -o /app

FROM scratch
COPY --from=builder /app /app
ENTRYPOINT ["/app"]
```

## 12. Типы ссылок

text

```
┌────────────────────────────────────────────────────────────┐
│                      ТИПЫ ССЫЛОК                           │
├──────────────────────────┬─────────────────────────────────┤
│     HARD LINK            │     SYMBOLIC LINK (symlink)     │
│     (жёсткая)            │     (символическая)             │
├──────────────────────────┼─────────────────────────────────┤
│ Дополнительное ИМЯ       │ Отдельный файл с ПУТЁМ к цели   │
│ для того же inode        │ (как ярлык в Windows)           │
│                          │                                 │
│ Удалить оригинал →       │ Удалить оригинал →              │
│ данные ОСТАЮТСЯ          │ ссылка ЛОМАЕТСЯ (dangling)      │
│ (пока есть хоть 1 ссылка)│                                 │
│                          │                                 │
│ ❌ Нельзя на директории  │ ✅ Можно на директории          │
│ ❌ Нельзя между ФС       │ ✅ Можно между ФС               │
│ Тот же inode             │ Свой inode                      │
└──────────────────────────┴─────────────────────────────────┘

Hard link:                  Symbolic link:

file_a ──┐                  symlink → /path/to/file_a
         ├──► inode 123          │
file_b ──┘    (данные)           ▼
                             file_a ──► inode 123
                                        (данные)
```

Bash

```
# Создать жёсткую ссылку
ln original.txt hardlink.txt

# Создать символическую ссылку
ln -s original.txt symlink.txt

# Символическая ссылка на директорию
ln -s /var/log ~/logs

# Посмотреть inode (у hard links одинаковый)
ls -li file.txt
# 12345 -rw-r--r-- 2 user group 1024 file.txt
# ^                ^
# inode            количество hard links

# Куда указывает symlink
readlink symlink.txt
readlink -f symlink.txt    # полный путь

# Найти сломанные symlinks
find /path -xtype l
```

## 14. Execute бит на директории

text

```
┌────────────────────────────────────────────────────────────┐
│              EXECUTE (x) НА ДИРЕКТОРИИ                     │
├────────────────────────────────────────────────────────────┤
│                                                            │
│  x на директории = право ВОЙТИ в неё и обращаться к       │
│  inode файлов внутри                                       │
│                                                            │
│  ┌───────────────┬────────────────────────────────────┐   │
│  │ Права на dir  │ Что можно делать                   │   │
│  ├───────────────┼────────────────────────────────────┤   │
│  │ r-- (4)       │ ls видит ИМЕНА, но нет доступа     │   │
│  │               │ к файлам внутри                    │   │
│  │ --x (1)       │ cd работает, cat file если знаешь  │   │
│  │               │ имя, НО ls НЕ работает             │   │
│  │ r-x (5)       │ Полноценный просмотр + доступ      │   │
│  │ -wx (3)       │ Создание/удаление, но без ls       │   │
│  └───────────────┴────────────────────────────────────┘   │
│                                                            │
│  Пример:                                                  │
│  chmod u-x testdir                                        │
│  cd testdir            # Permission denied                │
│  cat testdir/file.txt  # Permission denied                │
│                                                            │
└────────────────────────────────────────────────────────────┘
```

### Минимальные права для чтения файла

Bash

```
# Чтобы прочитать файл в директории нужно:
#
# На директорию: --x (1) — войти, обратиться к inode
# На файл:       r-- (4) — прочитать содержимое

chmod 100 directory/           # --x------
chmod 400 directory/file.txt   # r--------
cat directory/file.txt         # Работает!

# Если нужен ещё и ls (листинг):
chmod 500 directory/           # r-x------
```

## 21. Sticky bit, SUID, SGID

### SUID (Set User ID) — бит 4000

Bash

```
# При выполнении файл запускается с правами ВЛАДЕЛЬЦА файла,
# а не того, кто его запустил

ls -l /usr/bin/passwd
# -rwsr-xr-x 1 root root ... /usr/bin/passwd
#    │
#    └── s = SUID

# Любой пользователь запускает passwd →
# процесс работает от root →
# может писать в /etc/shadow

# Установить SUID
chmod u+s file
chmod 4755 file

# Убрать
chmod u-s file
```

### SGID (Set Group ID) — бит 2000

Bash

```
# НА ФАЙЛЕ: файл выполняется с правами ГРУППЫ-владельца

# НА ДИРЕКТОРИИ (основное применение):
# новые файлы НАСЛЕДУЮТ ГРУППУ директории
# (а не основную группу создателя)

# Пример: общая директория для команды
sudo mkdir /project
sudo chown root:developers /project
sudo chmod 2775 /project

ls -ld /project
# drwxrwsr-x 2 root developers ... /project
#       │
#       └── s = SGID

# Теперь все файлы в /project получат группу developers

# Установить SGID
chmod g+s directory/
chmod 2775 directory/
```

### Sticky bit — бит 1000

Bash

```
# НА ДИРЕКТОРИИ: удалять файлы может ТОЛЬКО:
# • владелец файла
# • владелец директории
# • root

# Даже если у пользователя есть w на директорию!

ls -ld /tmp
# drwxrwxrwt ... /tmp
#          │
#          └── t = sticky bit

# Все могут создавать файлы в /tmp,
# но удалить можно только СВОИ

# Установить sticky bit
chmod +t directory/
chmod 1777 directory/
```

### Сводная таблица

text

```
┌─────────┬───────┬──────────────────────┬────────────────────────┐
│   Бит   │ Число │ На файле             │ На директории          │
├─────────┼───────┼──────────────────────┼────────────────────────┤
│ SUID    │ 4000  │ Запуск от владельца  │ (не используется)      │
│ SGID    │ 2000  │ Запуск от группы     │ Наследование группы    │
│ Sticky  │ 1000  │ (не используется)    │ Удалять только своё    │
└─────────┴───────┴──────────────────────┴────────────────────────┘

Обозначение в ls -l:
• s вместо x — бит установлен + execute есть
• S вместо x — бит установлен, execute НЕТ (⚠️ ошибка)
• t вместо x — sticky + execute
• T вместо x — sticky, execute НЕТ
```

Bash

```
# Числовой формат с спецбитами (4 цифры)
chmod 4755 file    # SUID + rwxr-xr-x
chmod 2775 dir/    # SGID + rwxrwxr-x
chmod 1777 dir/    # Sticky + rwxrwxrwx

# Найти все SUID/SGID файлы (аудит безопасности!)
find / -perm -4000 -type f 2>/dev/null    # SUID
find / -perm -2000 -type f 2>/dev/null    # SGID
```

```
# ═══════════════════════════════════════════════════════════
#                         LVM
# ═══════════════════════════════════════════════════════════

# LVM = Logical Volume Manager
# Позволяет гибко управлять дисковым пространством

# Иерархия:
# Physical Volumes (PV) → Volume Groups (VG) → Logical Volumes (LV)
#
# Диски/разделы → Объединяются → Разбиваются на логические тома
# (/dev/sda1)       в группу       (которые форматируются и монтируются)

# ───────────────────────────────────────────────────────────
# Physical Volumes
# ───────────────────────────────────────────────────────────

# Создать PV
sudo pvcreate /dev/sdb
sudo pvcreate /dev/sdc

# Посмотреть PV
sudo pvs
sudo pvdisplay

# ───────────────────────────────────────────────────────────
# Volume Groups
# ───────────────────────────────────────────────────────────

# Создать VG из PV
sudo vgcreate myvg /dev/sdb /dev/sdc

# Посмотреть VG
sudo vgs
sudo vgdisplay

# Расширить VG новым диском
sudo vgextend myvg /dev/sdd

# ───────────────────────────────────────────────────────────
# Logical Volumes
# ───────────────────────────────────────────────────────────

# Создать LV
sudo lvcreate -L 10G -n mylv myvg           # Размер 10GB
sudo lvcreate -l 100%FREE -n mylv myvg      # Всё свободное место

# Посмотреть LV
sudo lvs
sudo lvdisplay

# Путь к LV
# /dev/myvg/mylv
# или
# /dev/mapper/myvg-mylv

# Форматировать
sudo mkfs.ext4 /dev/myvg/mylv

# Смонтировать
sudo mount /dev/myvg/mylv /mnt/mydata

# ───────────────────────────────────────────────────────────
# Изменение размера
# ───────────────────────────────────────────────────────────

# Расширить LV
sudo lvextend -L +5G /dev/myvg/mylv         # Добавить 5GB
sudo lvextend -l +100%FREE /dev/myvg/mylv   # Добавить всё свободное

# Расширить файловую систему
sudo resize2fs /dev/myvg/mylv               # Для ext4
sudo xfs_growfs /mnt/mydata                 # Для XFS

# Одной командой (расширить LV + ФС)
sudo lvextend -r -L +5G /dev/myvg/mylv

# Уменьшить LV (ОСТОРОЖНО! Сначала размонтировать и уменьшить ФС)
sudo umount /mnt/mydata
sudo e2fsck -f /dev/myvg/mylv
sudo resize2fs /dev/myvg/mylv 8G
sudo lvreduce -L 8G /dev/myvg/mylv

# ───────────────────────────────────────────────────────────
# Снапшоты
# ───────────────────────────────────────────────────────────

# Создать снапшот
sudo lvcreate -L 1G -s -n mysnapshot /dev/myvg/mylv

# Восстановить из снапшота
sudo lvconvert --merge /dev/myvg/mysnapshot

# Удалить снапшот
sudo lvremove /dev/myvg/mysnapshot
```
