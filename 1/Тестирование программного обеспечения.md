#### Пример Unit-теста

Python

```
# ═══════════════════════════════════════════════════════════
#                    UNIT-ТЕСТ (Python)
# ═══════════════════════════════════════════════════════════

# Код для тестирования
class Calculator:
    def add(self, a: int, b: int) -> int:
        return a + b
    
    def divide(self, a: int, b: int) -> float:
        if b == 0:
            raise ValueError("Division by zero")
        return a / b


# Unit-тесты
import pytest

class TestCalculator:
    """Тестируем ТОЛЬКО Calculator, без зависимостей"""
    
    def test_add_positive_numbers(self):
        calc = Calculator()
        result = calc.add(2, 3)
        assert result == 5
    
    def test_add_negative_numbers(self):
        calc = Calculator()
        result = calc.add(-2, -3)
        assert result == -5
    
    def test_divide_normal(self):
        calc = Calculator()
        result = calc.divide(10, 2)
        assert result == 5.0
    
    def test_divide_by_zero_raises_error(self):
        calc = Calculator()
        with pytest.raises(ValueError, match="Division by zero"):
            calc.divide(10, 0)
```

```
# Код для тестирования
class Database:
    def __init__(self):
        # Реальное подключение к БД
        self.connection = self._connect()
    
    def save_user(self, user):
        # SQL запрос
        pass
    
    def get_user(self, user_id):
        # SQL запрос
        pass


class UserService:
    def __init__(self, database: Database):
        self.db = database
    
    def create_user(self, name: str, email: str):
        user = {"name": name, "email": email}
        return self.db.save_user(user)


# Интеграционный тест
import pytest
from testcontainers.postgres import PostgresContainer

class TestUserServiceIntegration:
    """
    Тестируем ИНТЕГРАЦИЮ UserService и Database.
    Используем РЕАЛЬНУЮ БД (в контейнере).
    """
    
    @pytest.fixture
    def database(self):
        # Запускаем реальную PostgreSQL в Docker
        with PostgresContainer("postgres:15") as postgres:
            db = Database(postgres.get_connection_url())
            yield db
    
    def test_create_and_retrieve_user(self, database):
        """Тест взаимодействия сервиса и БД"""
        service = UserService(database)
        
        # Создаём пользователя (пишем в РЕАЛЬНУЮ БД)
        user_id = service.create_user("Alice", "alice@example.com")
        
        # Читаем из БД
        user = database.get_user(user_id)
        
        # Проверяем интеграцию
        assert user["name"] == "Alice"
        assert user["email"] == "alice@example.com"
```

```
const { test, expect } = require('@playwright/test');

test.describe('E2E: User Registration and Login', () => {
    
    test('user can register, login, and view profile', async ({ page }) => {
        // ШАГ 1: Регистрация
        await page.goto('https://example.com/register');
        
        await page.fill('input[name="email"]', 'test@example.com');
        await page.fill('input[name="password"]', 'SecurePass123');
        await page.fill('input[name="confirmPassword"]', 'SecurePass123');
        await page.click('button[type="submit"]');
        
        // Проверяем успешную регистрацию

        await expect(page.locator('.success-message')).toContainText(
            'Registration successful'
        );
        
        // ШАГ 2: Логаут
        await page.click('button.logout');
        
        // ШАГ 3: Логин
        await page.goto('https://example.com/login');
        await page.fill('input[name="email"]', 'test@example.com');
        await page.fill('input[name="password"]', 'SecurePass123');
        await page.click('button[type="submit"]');
        
        // ШАГ 4: Проверяем профиль
        await page.goto('https://example.com/profile');
        await expect(page.locator('.user-email')).toContainText(
            'test@example.com'
        );
        
        // Весь flow от регистрации до профиля протестирован!
    });
});
```

## 2. Классификация по методу тестирования

text

```
┌────────────────────────────────────────────────────────────┐
│           ЧЁРНЫЙ ЯЩИК vs БЕЛЫЙ ЯЩИК                        │
├────────────────────────────────────────────────────────────┤
│                                                            │
│  ЧЁРНЫЙ ЯЩИК (Black Box Testing)                          │
│  ────────────────────────────────                          │
│  • Тестируем БЕЗ знания внутренней реализации            │
│  • Фокус на входных данных и выходных результатах        │
│  • Не знаем КАК работает код внутри                      │
│                                                            │
│  ┌──────────────────┐                                     │
│  │                  │                                     │
│  │   ЧЁРНЫЙ ЯЩИК    │                                     │
│  │                  │                                     │
│  │    Не знаем      │                                     │
│  │    что внутри    │                                     │
│  │                  │                                     │
│  └──────────────────┘                                     │
│   Вход → [?????] → Выход                                 │
│                                                            │
│  БЕЛЫЙ ЯЩИК (White Box Testing)                           │
│  ───────────────────────────────                           │
│  • Тестируем СО знанием кода                             │
│  • Фокус на покрытии веток, циклов, условий              │
│  • Знаем КАК работает код                                │
│                                                            │
│  ┌──────────────────┐                                     │
│  │  if (x > 0) {    │                                     │
│  │    return "+"    │  ← Тестируем все ветки             │
│  │  } else {        │                                     │
│  │    return "-"    │                                     │
│  │  }               │                                     │
│  └──────────────────┘                                     │
│                                                            │
│  СЕРЫЙ ЯЩИК (Grey Box Testing)                            │
│  ──────────────────────────────                            │
│  • Частичное знание внутренней структуры                 │
│  • Комбинация чёрного и белого ящика                     │
│                                                            │
└────────────────────────────────────────────────────────────┘
```

```
// load-test.js
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Rate } from 'k6/metrics';

// Кастомная метрика
const errorRate = new Rate('errors');

// Конфигурация теста
export const options = {
    // Сценарий нагрузки
    stages: [
        { duration: '1m', target: 50 },   // Разгон до 50 польз. за 1 мин
        { duration: '3m', target: 50 },   // Держим 50 польз. 3 мин
        { duration: '1m', target: 100 },  // Разгон до 100 за 1 мин
        { duration: '3m', target: 100 },  // Держим 100 3 мин
        { duration: '1m', target: 0 },    // Плавное снижение
    ],
    
    // Пороговые значения (SLA)
    thresholds: {
        http_req_duration: ['p(95)<500'],  // 95% запросов < 500ms
        http_req_failed: ['rate<0.01'],    // < 1% ошибок
        errors: ['rate<0.05'],             // < 5% ошибок валидации
    },
};

// Сценарий теста
export default function () {
    // 1. Получить список пользователей
    const listResponse = http.get('https://api.example.com/users');
    
    check(listResponse, {
        'status 200': (r) => r.status === 200,
        'response time < 500ms': (r) => r.timings.duration < 500,
        'has users': (r) => JSON.parse(r.body).length > 0,
    }) || errorRate.add(1);
    
    sleep(1); // Think time
    
    // 2. Получить конкретного пользователя
    const userResponse = http.get('https://api.example.com/users/123');
    
    check(userResponse, {
        'status 200': (r) => r.status === 200,
        'has user data': (r) => JSON.parse(r.body).id === 123,
    }) || errorRate.add(1);
    
    sleep(Math.random() * 2); // Случайная пауза 0-2 сек
}

// Запуск: k6 run load-test.js
```

```
# locustfile.py
from locust import HttpUser, task, between
import random

class WebsiteUser(HttpUser):
    """Виртуальный пользователь сайта"""
    
    # Пауза между действиями (имитация реального пользователя)
    wait_time = between(1, 5)
    
    def on_start(self):
        """Выполняется один раз при старте пользователя"""
        # Логин
        self.client.post("/login", json={
            "username": "testuser",
            "password": "password123"
        })
    
    @task(3)  # Вес задачи: выполняется чаще
    def view_products(self):
        """Просмотр списка продуктов"""
        with self.client.get("/api/products", catch_response=True) as response:
            if response.status_code == 200:
                response.success()
            else:
                response.failure(f"Got status {response.status_code}")
    
    @task(2)
    def view_product_detail(self):
        """Просмотр деталей продукта"""
        product_id = random.randint(1, 100)
        self.client.get(f"/api/products/{product_id}")
    
    @task(1)
    def add_to_cart(self):
        """Добавление в корзину"""
        self.client.post("/api/cart", json={
            "product_id": random.randint(1, 100),
            "quantity": 1
        })
    
    @task(1)
    def search(self):
        """Поиск"""
        keywords = ["laptop", "phone", "tablet", "watch"]
        keyword = random.choice(keywords)
        self.client.get(f"/api/search?q={keyword}")
```
