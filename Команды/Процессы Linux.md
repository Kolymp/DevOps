```
# Просмотр процессов
ps aux                                # Все процессы
ps aux | grep nginx                   # Найти процесс
top                                   # Интерактивный мониторинг
htop                                  # Улучшенный top
pstree                                # Дерево процессов

# Управление процессами
kill <PID>                            # Завершить процесс (SIGTERM)
kill -9 <PID>                         # Принудительно завершить (SIGKILL)
killall nginx                         # Завершить все процессы nginx
pkill -f "python app.py"              # Завершить по имени

# Фоновые задачи
command &                             # Запустить в фоне
jobs                                  # Список фоновых задач
fg %1                                 # Вернуть задачу на передний план
bg %1                                 # Продолжить в фоне
nohup command &                       # Запустить с игнорированием SIGHUP
disown                                # Отвязать от терминала

# Системные ресурсы
free -h                               # Использование RAM
df -h                                 # Использование дисков
du -sh /path                          # Размер директории
du -h --max-depth=1 /path             # Размеры поддиректорий
iostat                                # I/O статистика
vmstat                                # Статистика виртуальной памяти
```
