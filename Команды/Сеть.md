```
# Проверка соединения
ping google.com                       # ICMP ping
ping -c 4 8.8.8.8                     # 4 пакета
curl https://example.com              # HTTP запрос
curl -I https://example.com           # Только заголовки
wget https://example.com/file.zip     # Скачать файл

# Информация о сети
ip addr                               # IP-адреса интерфейсов
ip route                              # Таблица маршрутизации
ifconfig                              # Старый способ (deprecated)
netstat -tuln                         # Прослушиваемые порты
ss -tuln                              # Современная альтернатива netstat
lsof -i :80                           # Кто использует порт 80

# DNS
nslookup google.com                   # DNS-запрос
dig google.com                        # Детальная DNS-информация
host google.com                       # Простой DNS-запрос

# Трассировка
traceroute google.com                 # Маршрут до хоста
mtr google.com                        # Комбинация ping + traceroute

# Файрвол (firewalld)
firewall-cmd --list-all               # Текущие правила
firewall-cmd --add-port=80/tcp --permanent
firewall-cmd --reload

# Файрвол (ufw - Ubuntu)
ufw status                            # Статус
ufw allow 22/tcp                      # Разрешить SSH
ufw enable                            # Включить
```
